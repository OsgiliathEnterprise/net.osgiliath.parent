75    /**
76     * probe adding the abstract test class.
77     * 
78     * @param builder
79     *          the pax probe builder
80     * @return the provisionned probe.
81     */
82    @ProbeBuilder
83    public TestProbeBuilder extendProbe(TestProbeBuilder builder) {
84      builder.addTest(AbstractPaxExamKarafConfiguration.class);
85      builder.setHeader(Constants.EXPORT_PACKAGE,
86          "itests.net.osgiliath.feature.itest.validation.cdi");
87      builder.setHeader(Constants.BUNDLE_MANIFESTVERSION, "2");
88      builder.setHeader(Constants.DYNAMICIMPORT_PACKAGE, "*");
89      return builder;
90    }
91  
92    /**
93     * Send null message to trigger a validation exception.
94     * 
95     * @throws Exception
96     *           the expected violation
97     */
98    @Test(expected = ConstraintViolationException.class)
99    public void testValidateNull() throws ConstraintViolationException {
100     if (LOG.isDebugEnabled()) {
101       for (final Bundle b : this.bundleContext.getBundles()) {
102         LOG.debug("bundle: " + b.getSymbolicName() + ", state: " + b.getState());
103       }
104     }
105     try {
106       this.consumer.nullMessageValidation(null);
107       fail("Tho shall not be here");
108     }
109     catch (IllegalArgumentException iae) {
110       if (LOG.isDebugEnabled()) {
111         LOG.debug("Expected null check trace", iae);
112       }
113     }
114     final HelloObject object = new HelloObject();
115     object.setMessage(null);
116     this.consumer.nullMessageValidation(object);
117     fail("Tho shall not be here");
118 
119   }
