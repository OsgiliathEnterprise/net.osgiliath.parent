35   * sample of an integration test

37   * @author charliemordant

39   */
40  public class ITHelloWebUITest {
41    private static Logger LOG = LoggerFactory.getLogger(ITHelloWebUITest.class);
42  
43    /**
44     * . A correct {@link AbstractUtilisateur} Name
45     */
46    private static final String CORRECT_NAME = "CorrectName";
47  
48    @Test
49    
50    public void testSayHello() throws Exception {
51      final WebDriver driver;
52      driver = new FirefoxDriver();
53  
54      // Sleep until the elements we want is visible or 5 seconds is over
55      long end = System.currentTimeMillis() + 10000;
56         (new WebDriverWait(driver, 300)).until(new ExpectedCondition<Boolean>() {
57              public Boolean apply(WebDriver d) {
58                driver.get("http://localhost:8181/net.osgiliath.sample.webapp.ui");
59                boolean ret = d.getTitle().toLowerCase().startsWith("hello");
60                if (!ret) {
61                  driver.navigate().refresh();
62                }
63                return ret;
64              }
65            });
66      WebElement element;
67      element = driver.findElement(By.id("helloInput"));
68      element.click();
69      element.sendKeys(CORRECT_NAME);
70      element = driver.findElement(By.id("helloButtonAsync"));
71      element.click();
72      (new WebDriverWait(driver, 20)).until(new ExpectedCondition<Boolean>() {
73        public Boolean apply(WebDriver d) {
74  
75          Collection<WebElement> cells = d.findElements(By
76              .xpath("//table//tr//td"));
77  
78          return cells.size() == 1;
79        }
80      });
81      end = System.currentTimeMillis() + 10000;
82  
83      driver.close();
84    }
85  
86  }
