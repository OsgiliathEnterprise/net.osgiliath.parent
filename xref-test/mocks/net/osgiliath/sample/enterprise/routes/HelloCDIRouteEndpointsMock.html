23  import javax.xml.bind.JAXBContext;
24  import net.osgiliath.sample.webapp.business.spi.model.Hellos;
25  import net.osgiliath.sample.webapp.model.entities.HelloEntity;
26  import org.apache.camel.Exchange;
27  import org.apache.camel.builder.RouteBuilder;
28  import org.apache.camel.converter.jaxb.JaxbDataFormat;
29  import org.apache.camel.spi.DataFormat;
30  
31  public class HelloCDIRouteEndpointsMock extends RouteBuilder {
32    // TODO Mock routing, you can also use integration test like in the business
33    // module
34    @Override
35    public void configure() throws Exception {
36      JAXBContext ctx = JAXBContext.newInstance(new Class[] { HelloEntity.class,
37          Hellos.class });
38      // initialize jaxbformat from jaxbcontext
39      DataFormat jaxBDataFormat = new JaxbDataFormat(ctx);
40      from("properties:{{helloApp.restEndpoint}}").choice()
41          .when(header(Exchange.HTTP_METHOD).isEqualTo(constant("GET")))
42          .beanRef("helloService", "getHellos").marshal(jaxBDataFormat)
43          .when(header(Exchange.HTTP_METHOD).isEqualTo(constant("POST")))
44          .unmarshal(jaxBDataFormat)
45          .beanRef("helloService", "persistHello(${body})").endChoice();
46    }
47  
48  }
