23  import com.google.common.collect.Iterables;
24  import com.google.common.collect.Lists;
25  import java.util.stream.Collectors;
26  import javax.inject.Inject;
27  import net.osgiliath.sample.webapp.business.spi.annotations.REST;
28  import net.osgiliath.sample.webapp.business.spi.model.Hellos;
29  import net.osgiliath.sample.webapp.model.daos.HelloRepository;
30  import net.osgiliath.sample.webapp.model.entities.HelloEntity;
31  import org.ops4j.pax.cdi.api.OsgiService;
32  
33  
34  /**
35   * REST Service implementation for testing purpose.
36   * 
37   * @author charliemordant
38   */
39  @REST
40  public class HelloServiceImpl implements HelloServiceJaxRS {
41  	
42  	/**
43  	 * JPA repository.
44  	 */
45  	@Inject
46  	@OsgiService
47  	private HelloRepository repository;
48  	
49    
50  
51    /**
52     * Registering instance.
53     * 
54     * @param helloObject
55     *          the element to save in the database
56     */
57    @Override
58    public void persistHello(final HelloEntity helloObject) {
59  	  repository.save(helloObject);
60    }
61  
62    /**
63     * Returns registered instances.
64     * 
65     * @return all instances
66     */
67    @Override
68    public Hellos getHellos() {
69      
70      return new Hellos(repository.findAll().stream().map(input -> input.getHelloMessage()).collect(Collectors.toList()));
71    }
72  
73    /**
74     * Deletes all elements.
75     */
76    @Override
77    public void deleteHellos() {
78  	  repository.deleteAll();
79      
80    }
81  
82  
83  }
