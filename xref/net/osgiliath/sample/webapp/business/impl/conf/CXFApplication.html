23  import com.fasterxml.jackson.jaxrs.json.JacksonJsonProvider;
24  import com.google.common.collect.Sets;
25  import io.swagger.jaxrs.listing.SwaggerSerializers;
26  import java.util.Set;
27  import javax.inject.Inject;
28  import javax.ws.rs.ApplicationPath;
29  import javax.ws.rs.core.Application;
30  import lombok.extern.slf4j.Slf4j;
31  import net.osgiliath.module.cdi.eager.Eager;
32  import net.osgiliath.sample.webapp.business.impl.rest.HelloServiceJaxRS;
33  import net.osgiliath.sample.webapp.business.spi.HelloService;
34  import net.osgiliath.sample.webapp.business.spi.annotations.REST;
35  import org.apache.cxf.jaxrs.provider.JAXBElementProvider;
36  import org.apache.cxf.jaxrs.swagger.Swagger2Feature;
37  import org.apache.cxf.jaxrs.validation.ValidationExceptionMapper;
38  
39  /**
40   * Main CXF application that publishes services.
41   * 
42   * @author charliemordant
43   *
44   */
45  @Eager
46  @ApplicationPath("/helloService")
47  @Slf4j
48  public class CXFApplication extends Application {
49    /**
50     * The injected REST endpoint.
51     */
52    @Inject
53    @REST
54    private HelloService helloService;
55    /**
56     * Swagger support.
57     */
58    @Inject
59    private SwaggerAPIAccessService swagger;
60  
61    /**
62     * @see {@link javax.ws.rs.core.Application#getSingletons()}
63     */
64    @Override
65    public Set<Object> getSingletons() {
66      log.error("Osgiliath: registering cxf servlet");
67      Swagger2Feature feature = new Swagger2Feature();
68  
69      // customize some of the properties
70      feature.setBasePath("/api");
71      feature.setResourcePackage(HelloServiceJaxRS.class.getPackage().getName());
72      return Sets.<Object> newHashSet(helloService, swagger,
73           new JAXBElementProvider<Object>(),
74          // new ExceptionXmlMapper(),
75  
76          new SwaggerSerializers(), new ValidationExceptionMapper(),
77          // new JAXRSBeanValidationInInterceptor(),
78          // new JAXRSBeanValidationOutInterceptor(),
79  
80          new JacksonJsonProvider());
81    }
82  }
