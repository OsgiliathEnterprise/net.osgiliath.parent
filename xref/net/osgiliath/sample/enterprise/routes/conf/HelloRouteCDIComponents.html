25  import javax.enterprise.inject.Produces;
26  import javax.inject.Inject;
27  import javax.inject.Named;
28  import lombok.extern.slf4j.Slf4j;
29  import net.osgiliath.module.camel.ThrownExceptionMessageToInBodyProcessor;
30  import net.osgiliath.module.cdi.eager.Eager;
31  import org.apache.camel.Component;
32  import org.apache.camel.Processor;
33  import org.apache.camel.component.http.HttpComponent;
34  import org.apache.camel.component.jackson.JacksonDataFormat;
35  import org.apache.camel.component.properties.DefaultPropertiesParser;
36  import org.apache.camel.component.properties.PropertiesComponent;
37  import org.apache.camel.component.properties.PropertiesParser;
38  import org.apache.camel.dataformat.xmljson.XmlJsonDataFormat;
39  import org.apache.camel.spi.DataFormat;
40  import org.apache.deltaspike.core.api.config.ConfigResolver;
41  import org.ops4j.pax.cdi.api.OsgiService;
42  
43  /**
44   * CDI configuration.
45   * 
46   * @author charliemordant
47   */
48  @Slf4j
49  @Eager
50  public class HelloRouteCDIComponents {
51    /**
52     * Messaging component import.
53     */
54    @Inject
55    @OsgiService(filter = "(component-type=jms)", dynamic = true)
56    private Component jms;
57    /**
58     * transactional messaging component.
59     */
60    @Inject
61    @OsgiService(filter = "(component-type=jmsXA)", dynamic = true)
62    private Component jmsXA;
63  
64    /**
65     * Exception message converter processor.
66     * 
67     * @return JSR303 message processor
68     */
69    @Produces
70    @Named("thrownExceptionMessageToInBodyProcessor")
71    public Processor getThrownExceptionMessageToInBodyProcessor() {
72      log.info("Inject Processor in body");
73      return new ThrownExceptionMessageToInBodyProcessor();
74    }
75  
76    /**
77     * JMS Component.
78     * 
79     * @return JMS component CDI export
80     */
81    @Produces
82    @Named("jms")
83    public Component getJms() {
84      log.info("Inject jms");
85      return this.jms;
86    }
87  
88    /**
89     * JMS XA component.
90     * 
91     * @return JMS XA component CDI export
92     */
93    @Produces
94    @Named("jmsXA")
95    public Component getJmsXA() {
96      log.info("Inject jmsXA");
97      return this.jmsXA;
98    }
99  
100   /**
101    * HTTP camel component.
102    * 
103    * @return HTTP component CDI export
104    */
105   @Produces
106   @Named("http")
107   public Component getHttp() {
108     log.info("Inject httpComponent");
109     return new HttpComponent();
110   }
111 
112   /**
113    * JSon dataformat.
114    * 
115    * @return Json component CDI export
116    */
117   @Produces
118   @Named("json")
119   public DataFormat getJacksonDataFormat() {
120     return new JacksonDataFormat();
121   }
122 
123   /**
124    * XML to json dataformat.
125    * 
126    * @return XmlJson component CDI export
127    */
128   @Produces
129   @Named("xmljson")
130   public DataFormat getXmlJsonDataFormat() {
131     return new XmlJsonDataFormat();
132   }
133 
134   @Produces
135   @Named("properties")
136   PropertiesComponent properties(PropertiesParser parser) {
137     PropertiesComponent component = new PropertiesComponent();
138     component.setPropertiesParser(parser);
139     return component;
140   }
141 
142   // PropertiesParser bean that uses DeltaSpike to resolve properties
143   static class DeltaSpikeParser extends DefaultPropertiesParser {
144     @Override
145     public String parseProperty(String key, String value,
146         Properties properties) {
147       return ConfigResolver.getPropertyValue(key);
148     }
149   }
150 }
